```
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
    int n,len;
    vector<string> vec,vecs;
    vector< vector<string> > vecvs;
    while(scanf("%d",&n)!=EOF){
        string str;
        if(n==0)
            break;
        vec.clear();
        //vecs.clear();
        vecvs.clear();
        for(int i=0;i<n;i++){
            cin>>str;
            len=str.size();
            if(str[len-1]!='\\')
                str+='\\';
            ///cout<<len<<" "<<str<<endl;
            vec.push_back(str);
        }
        /*
        for(int i=0;i<n;i++){
            cout<<vec[i]<<endl;
        }
        */

        sort(vec.begin(),vec.end());

        for(int i=0;i<vec.size();i++){
            int st=0,ed=0;
            vecs.clear();///每个一维数组初始化
            while(ed!=string::npos){
                ed=vec[i].find('\\',st);
                //cout<<vec[i].substr(st,ed-st)<<" ";
                vecs.push_back(vec[i].substr(st,ed-st));
                st=ed+1;
            }
            //cout<<endl;
            //for(int j=0;j<vecs.size();j++){
            //    cout<<vecs[j]<<" ";
            //}
            //cout<<endl;
            vecvs.push_back(vecs);///存放进二维数组中
        }

        for(int i=0;i<n;i++){
            if(i>0&&vecvs[i][0]==vecvs[i-1][0]){
            ///与前一个目录有相同的根目录
                int j=1;
                while(j<vecvs[i].size()&&vecvs[i][j]==vecvs[i-1][j])
                ///找到第一个不同的目录
                    j++;
                for(;j<vecvs[i].size()-1;j++){
                    for(int k=0;k<j;k++)
                        cout<<" ";
                    cout<<vecvs[i][j]<<endl;
                }
            }
            else{
                for(int j=0;j<vecvs[i].size()-1;j++){
                    for(int k=0;k<j;k++)
                        cout<<" ";
                    cout<<vecvs[i][j]<<endl;
                }
            }
            //cout<<i<<endl;
        }

        //cout<<i<<endl;

    }


    return 0;
}
```
