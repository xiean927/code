#include <bits/stdc++.h>

using namespace std;

bool isprime(int n){
    if(n<=1)    return false;
    int sqr=sqrt(1.0*n);
    for(int i=2;i<=sqr;i++)
        if(n%i==0)  return false;
    return true;
}

int gcd(int a,int b){
    if(b==0)    return a;
    else    return gcd(b,a%b);
}

///计算数每位数字之和
int digitSum(int x){
    int sum=0;

    string s=to_string(x);
    for(int i=0,len=s.length();i<len;++i)
        sum+=s[i]-'0';
    return sum;
}

struct record{
    int sum,val;///val是数,sum是数的各位组成的和

    record(int v,int n):val(v),sum(n){}

    bool operator<(record &x){//为什么不是先按照val排序?而是按照sum排序（也就是最后答案的n,唯一吧?）
        if (sum!=x.sum)
            return sum<x.sum;
        else
            return val<x.val;
    }
};

vector<record> r;
/*
sum:数位和
val:组成的数字
left:剩余未分配的数字
target:目标数位和

*/
void dfs(int sum,int val,int left,int target){
    if(left==0&&sum==target){
        int n=digitSum(val+1),g=gcd(sum,n);
        if(g>2&&isprime(g)) ///如果val和val+1的最大公约数是素数,且大于2
            r.push_back(record(val,n));///则该数字可以
    } else if(left>0)///分配完剩余未分配的部分
        for (int i=0;i<=9;++i)///第left位从0~9枚举
		///当A余下的位都是9,组成的数是否大于等于m（取等，是因为A+1）,
            if(sum+i+left*9-9>=target&&sum+i<=target)
            ///数位和+i+剩余的（最大为(left-1)*9）的值大于目标数位和，
            ///且 数位和+i小于目标数位和，如果大于，则超限
                dfs(sum+i,val*10+i,left-1,target);
}

int main(){
    int n,k,m;
    scanf("%d",&n);
    for (int i=1;i<=n;++i){
        r.clear();
        scanf("%d%d",&k,&m);
        printf("Case %d\n",i);
        for (int j=1;j<=9;++j) dfs(j,j,k-1,m);///枚举第k-1位的值为j,
        if (r.empty())
            printf("No Solution\n");
        else {
            sort(r.begin(), r.end());
            for (int j=0;j<r.size();++j)
                printf("%d %d\n",r[j].sum,r[j].val);
        }
    }
}

### 3 Postfix Expression
#### 中缀表达式节点应该有三种状态
* 1.左右孩子都为空
* 2.左右孩子都非空
* 3.左孩子为空，右孩子非空


```
#include <bits/stdc++.h>

using namespace std;

struct node
{
	char data[15];
	int left,right;
}tree[25];

int view[25],root;

void InOrder(int v)
{
    if(v==-1)
        return ;
    printf("(");
    if(tree[v].left!=-1&&tree[v].right!=-1){
        InOrder(tree[v].left);
        InOrder(tree[v].right);
        printf("%s",tree[v].data);
    }else if(tree[v].left==-1&&tree[v].right!=-1){
        printf("%s",tree[v].data);
        InOrder(tree[v].right);
    }else if(tree[v].right==-1&&tree[v].left!=-1){//这个判断可以去掉
        InOrder(tree[v].left);
    }else{///左右子树均为空
        printf("%s",tree[v].data);
    }
    printf(")");
}
int main()
{
	int n ;
	scanf("%d",&n);
	for(int i = 1; i <= n; i++)
	{
		scanf("%s %d %d",tree[i].data,&tree[i].left, &tree[i].right);
		view[tree[i].left] = 1;
		view[tree[i].right] = 1;
	}
	for(int i = 1; i <= n; i++)
	{
		if(view[i] == 0)
		{
			root = i ;
			break;
		}
	}
	InOrder(root);
	return 0;
}

### 4 Dijkstra Sequence (30 分) 
```
//得分30,满分30
#include<iostream>
using namespace std;
const int INF=0x3f3f3f3f;
const int maxn=1005;
int d[maxn],s[maxn],G[maxn][maxn];
bool visit[maxn];
int main(){
    int n,m,a,b,c,k;
    scanf("%d %d\n",&n,&m);
    fill(G[0],G[0]+maxn*maxn,INF);
    while(m--){
        scanf("%d %d %d\n",&a,&b,&c);
        G[a][b]=G[b][a]=c;
    }
    scanf("%d\n",&k);
    while(k--){
        m=n;
        for(int i=1;i<=n;i++){
            scanf("%d",&s[i]);
        }
        bool f=true;
        int start=s[1];
        fill(visit,visit+maxn,false);
        fill(d,d+maxn,INF);
        d[start]=0;
        for(int i=1;i<=n;i++){
            int u = -1, minn = INF;
            for (int j = 1; j <= n; j++) {
                if (!visit[j] && d[j] < minn) {
                    u = j;
                    minn = d[j];
                }
            }
            if (u == -1 || d[u] != d[s[i]]){
                f=false;
                continue;
            }
            u = s[i];
            visit[u]=true;
            for(int v=1;v<=n;v++){
                if(!visit[v] && G[u][v]!=INF && G[u][v]+d[u]<d[v]){
                    d[v]=G[u][v]+d[u];
                }
            }
        }
        if(!f)printf("No\n");
        else printf("Yes\n");
    }
    return 0;
}
