```
#include <bits/stdc++.h>

using namespace std;

int k,m;

bool isprime(int n){
    if(n<=1)    return false;
    int sqr=sqrt(1.0*n);
    for(int i=2;i<=sqr;i++)
        if(n%i==0)  return false;
    return true;
}

int gcd(int a,int b){
    if(b==0)    return a;
    else    return gcd(b,a%b);
}

///计算数每位数字之和
int digitSum(int x){
    int sum=0;

    string s=to_string(x);
    int len=s.size();
    for(int i=0;i<len;++i)
        sum+=s[i]-'0';

    return sum;
}

struct record{
    int val,sum;///val是数,sum是数的各位组成的和
};

vector<record> r;

bool cmp(const record &a,const record &b){
    if(a.sum!=b.sum)
        return a.sum<b.sum;
    else
        return a.val<b.val;
    //if(a.val!=b.val)
    //return a.val<b.val;
}

/*
sum:数位和
val:组成的数字
left:剩余未分配的数字
m:目标数位和
*/
void dfs(int sum,int val,int left){
    if(left==0&&sum==m){
        int n=digitSum(val+1),g=gcd(sum,n);
        if(g>2&&isprime(g)){ ///如果val和val+1的最大公约数是素数,且大于2
            record tmpr;
            tmpr.val=val;
            tmpr.sum=n;
            r.push_back(tmpr);///则该数字可以
        }
    }
    else if(left>0){///分配完剩余未分配的部分
        for (int i=0;i<=9;++i)///第left位从0~9枚举
        ///当A余下的位都是9,组成的数是否大于等于m（取等，是因为A+1）,
            if(sum+i+left*9-9>=m&&sum+i<=m)
            ///数位和+i+剩余的（最大为(left-1)*9）的值大于目标数位和m，
            ///且 数位和+i小于目标数位和，如果大于，则剪枝
                dfs(sum+i,val*10+i,left-1);
    }
}

int main(){
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;++i){
        r.clear();
        scanf("%d%d",&k,&m);
        printf("Case %d\n",i);
        for(int j=1;j<=9;++j)
            dfs(j,j,k-1);///枚举第k-1位的值为j,
        if(r.empty())
            printf("No Solution\n");
        else{
            sort(r.begin(),r.end(),cmp);
            int lenr=r.size();
            for (int j=0;j<lenr;++j)
                printf("%d %d\n",r[j].sum,r[j].val);
        }
    }
}
```

```
#include <bits/stdc++.h>

using namespace std;

const int maxn = 100010;
struct node{
	int addr,v,next;
}Node[maxn];

int add1,add2,n,tempadd;

int main(){
	scanf("%d%d%d",&add1,&add2,&n);
	for(int i=0;i<n;i++){
		scanf("%d",&tempadd);
		Node[tempadd].addr = tempadd;
		scanf("%d%d",&Node[tempadd].v,&Node[tempadd].next);
	}

	vector<node> l1,l2;///长链表 l1;短链表 l2
	vector<node> temp,ans;
	while(add1!=-1){
		l1.push_back(Node[add1]);
		add1=Node[add1].next;
	}

	while(add2!=-1){
		l2.push_back(Node[add2]);
		add2=Node[add2].next;
	}

	///保证长链表至少是短链表的2倍长
	if(l2.size()>=2*l1.size()){///l1是包含结点多的那个链表,l2是包含结点少的那个链表
		temp=l1;
		l1=l2;
		l2=temp;
	}

	int k=0,index=l2.size()-1;

	int lenl=l1.size();
	for(int i=0;i<lenl;i++){
		ans.push_back(l1[i]);///推入长链表
		if((i+2+k)%3==0&&index>=0){///多加一个判断index>=0,短链表有效
            //cout<<"index:"<<index<<endl;
			ans.push_back(l2[index--]);///推入短链表
			k++;
		}
	}
	//cout<<"index:"<<index<<endl;
	while(index>=0)
        ans.push_back(l2[index--]);///短链表剩余的结点放到最后

    int lena=ans.size();
	for(int i=0;i<lena;i++){
		if(i!=lena-1)
            printf("%05d %d %05d\n",ans[i].addr,ans[i].v,ans[i+1].addr);
		else
            printf("%05d %d -1\n",ans[i].addr,ans[i].v);
	}

	return 0;
}
```

### 4 Dijkstra Sequence (30 分) 
```
#include <bits/stdc++.h>

using namespace std;

const int inf=0x3f3f3f3f;
const int maxn=1010;

int e[maxn][maxn];
int d[maxn],s[maxn];
bool vis[maxn];
int n,m;

int main(){

    scanf("%d%d",&n,&m);

    fill(e[0],e[0]+maxn*maxn,inf);

    int a,b,c,k;

    for(int i=0;i<m;i++){
        scanf("%d%d%d",&a,&b,&c);
        e[a][b]=e[b][a]=c;
    }

    scanf("%d",&k);

    while(k--){
        for(int i=1;i<=n;i++)
            scanf("%d",&s[i]);

        bool flag=true;
        int st=s[1];///起点
        fill(vis,vis+maxn,false);
        fill(d,d+maxn,inf);
        d[st]=0;
        for(int i=1;i<=n;i++){
            int u=-1,minn=inf;
            for(int j=1;j<=n;j++){
                if(!vis[j]&&d[j]<minn){
                    u=j;
                    minn=d[j];
                }
            }
            if(u==-1||d[u]!=d[s[i]]){///所寻找结点不是给定结点,返回false
                flag=false;
                break;
            }
            vis[u]=true;
            for(int v=1;v<=n;v++){
                if(!vis[v]&&e[u][v]!=inf&&d[v]>d[u]+e[u][v])
                    d[v]=e[u][v]+d[u];
            }
        }
        if(flag)
            printf("Yes\n");
        else
            printf("No\n");
    }

    return 0;

}
```
