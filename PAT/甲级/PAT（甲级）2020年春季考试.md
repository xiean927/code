

### 7-2 The Judger (25分)

* ```the difference of two numbers```意思是 两数之差
* 符合条件的数：
  * 1，是之前两数的差值
  * 2，是之前未出现过的数
  
  
```
#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int exist[100010]={0}, diff[100010] = {0}; 
// exist记录已经报过的数，diff记录已经报过的数的差值
vector<int> exists;
// exists记录已经报过的数，对于每个新报的数，和每个已存的数相减得差
int ht[15] = {0};
int a[1010][1010];


int main() {
		
    int n1, n2;
    scanf("%d%d", &n1, &n2);

    diff[abs(n1 - n2)] = 1;
    exist[n1] = 1;
    exist[n2] = 1;
    exists.push_back(n1);
    exists.push_back(n2);
    
    int n, m;
    scanf("%d%d", &n, &m);
    
    for(int i = 1; i <= n; i++) 
    	for(int j = 1; j <= m; j++) 
    		scanf("%d", &a[i][j]); 
    		
	for(int j = 1; j <= m; j++) {
		for(int i = 1; i <= n; i++) {
			
			if(ht[i] == 1) continue;///已经淘汰的人不再计数

    		int t = a[i][j];

    		if(diff[t] && !exist[t]) {///差值存在，且该数未被叫过
    			
    			for(int i = 0; i < exists.size(); i++) {
    				int t2 = exists[i];
    				diff[abs(t - t2)] = 1;///新增差值，
    			}
    			exist[t] = 1;
    			exists.push_back(t);
    			
    		} else {
    			ht[i] = 1;
    			printf("Round #%d: %d is out.\n", j, i);
    		}
		}
	}

    int f = 1;
    
    for(int i = 1; i <= n; i++) {
    	if(ht[i] == 0) {
    		if(f) printf("Winner(s):");
    		printf(" %d", i);
    		f = 0;
    	}
    }
    
    if(f) printf("No winner.");

    return 0;
}

```



### 7-4 Replacement Selection (30分)

* 外排序
* 两个梯队
  * 第一梯队，即将入队的元素（应属于 第一梯队），大于即将出队的元素
  * 第二梯队，即将入队的元素（应属于 第二梯队），小于等于即将出队的元素
  
  
```
#include <bits/stdc++.h>

using namespace std;

map<int,int> fmp,smp;
vector<vector<int> > ans;
vector<int> tmp;

int n,m;

int main(){

    scanf("%d%d",&n,&m);

    int num;
    for(int i=0;i<m;i++){
        scanf("%d",&num);
        fmp[num]++;
    }
    for(int i=m;i<n;i++){
        scanf("%d",&num);
        map<int,int>::iterator it=fmp.begin();
        int head=it->first;
        ///出队
        if(it->second==1)   fmp.erase(it);
        else    fmp[head]--;

        tmp.push_back(head);///进入第一梯队
        if(num > head)///大于第一梯队即将出队的元素
            fmp[num]++;
        else
            smp[num]++;

        if(fmp.size()==0){///如果第一梯队的元素输出完毕，
            fmp=smp;///将第二梯队转为第一梯队
            smp.clear();
            ans.push_back(tmp);
            tmp.clear();
        }
        if(i==n-1){///到了最后一个元素
            while(fmp.size()){///第一梯队，的元素全部输出
                it=fmp.begin();
                head=it->first;
                tmp.push_back(head);
                if(it->second==1)
                    fmp.erase(it);
                else
                    fmp[head]--;
            }
            if(tmp.size()>0)///如果第一梯队还有元素，就重新建立一队
                ans.push_back(tmp);
            tmp.clear();

            while(smp.size()){///第二梯队，的元素全部输出
                it=smp.begin();
                head=it->first;
                tmp.push_back(head);
                if(smp.begin()->second==1)
                    smp.erase(it);
                else
                    smp[head]--;
            }
            ans.push_back(tmp);
        }
    }
    for (int i = 0; i < (int)ans.size(); i++) {
		for (int j = 0; j < (int)ans[i].size(); j++) {
			if (j == 0)
                printf("%d", ans[i][j]);
			else
                printf(" %d", ans[i][j]);
			if (j == (int)ans[i].size() - 1)
                printf("\n");
		}
	}

    return 0;
}
```
