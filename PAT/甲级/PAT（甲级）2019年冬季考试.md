```
#include <bits/stdc++.h>

string a[26][7],str;
vector<int> word;

        for(int j=0;j<7;j++){

        }
    }

    getline(cin,str);
    bool flag=false;

    for(int i=0;i<len;i++){
        if(isupper(str[i]) && i==len-1)///到了最后一个字母
            word.push_back(str[i]-'A');
        ///以下是控制输出
        if(isupper(str[i]) && i!=len-1){///未到最后一个字母
            word.push_back(str[i]-'A');
        }else{
            if(flag)///第一次不换行,技巧1，
                printf("\n\n");

                int lenw=word.size();
                for(int j=0;j<lenw;j++){///一行有多少个字母，

                    cout<<a[word[j]][k];
                }
                if(k!=6)///控制每一行的回车输出
                    printf("\n");
            }
            word.clear();
            flag=true;
        }
    }
    return 0;
}
```

#### ```7-3 Summit (25分) ```AC
* 类似题目：```1142 Maximal Clique (25分)```
```

#include <bits/stdc++.h>

using namespace std;

int e[210][210];
int tmp[210];
int n,m,k;
set<int> st;

int main(){
    scanf("%d%d",&n,&m);
    int a,b;
    memset(e,0,sizeof(e));
    for(int i=0;i<m;i++){
        scanf("%d%d",&a,&b);
        e[a][b]=e[b][a]=1;
    }
    scanf("%d",&k);
    int cnt;
    bool flag;
    for(int i=1;i<=k;i++){
        scanf("%d",&cnt);
        st.clear();
        for(int j=0;j<cnt;j++){
            scanf("%d",&tmp[j]);
            st.insert(tmp[j]);
        }
        flag=true;
        for(int j=0;j<cnt;j++){
            for(int v=0;v<cnt;v++){
                if(v==j)    continue;///是节点本身，跳过
                else{
                    if(!e[tmp[j]][tmp[v]]){
                    ///如果存在组内的人不是组内的其他人的直接朋友,则直接跳出循环
                        flag=false;
                        break;
                    }
                }
            }
            if(!flag)
                break;
        }
        if(!flag)///如果存在组内的人不是组内的其他人的直接朋友,
            printf("Area %d needs help.\n",i);
        else{
            int tmpcnt=0;///与组内所有人都是直接朋友的人的人数
            int minindex=300;///一个与组内所有人都是直接朋友的人的最小编号
            for(int j=1;j<=n;j++){///遍历n个人，筛选出，一个与组内所有人都是直接朋友的人
                if(st.find(j)!=st.end())   ///如果发现了说明已经在圈内，continue 
                    continue;
                bool tmpf=true;
                for(int v=0;v<cnt;v++)
                    if(e[j][tmp[v]]==0)///第j号人，与这个圈子里的某一个人不是朋友，则第j号人不能被加入组内
                        tmpf=false;
                if(tmpf){///第j号人，与这个圈子里所有人都是朋友
                    tmpcnt++;
                    minindex=min(minindex,j);
                }
            }
            if(tmpcnt)
                printf("Area %d may invite more people, such as %d.\n",i,minindex);
            else
                printf("Area %d is OK.\n",i);
        }
    }
    return 0;
}
```

#### ```7-4 Cartesian Tree (30分) ```AC
* 每次遍历树中最小结点作为根,再建树

```
#include <bits/stdc++.h>

using namespace std;

struct Node{
    int data;
    Node *lchild,*rchild;
};

int num[40];
int n;

Node* create(int inL,int inR){
    if(inL>inR)
        return NULL;
    int rooti,minn=0x3f3f3f3f;
    for(int i=inL;i<=inR;i++){///每次寻找值最小的结点作为根，建树
        if(minn>num[i]){
            minn=num[i];
            rooti=i;
        }
    }
    Node* root=new Node;
    root->data=num[rooti];
    root->lchild=create(inL,rooti-1);
    root->rchild=create(rooti+1,inR);
    return root;///返回根，而非NULL
}

void InOrder(Node *root){
    if(root==NULL)
        return ;
    InOrder(root->lchild);
    printf("%d-",root->data);
    InOrder(root->rchild);
}

int main(){

    scanf("%d",&n);

    for(int i=0;i<n;i++){
        scanf("%d",&num[i]);
    }

    Node* root=create(0,n-1);
    queue<Node*> q;
    vector<int> v;

    q.push(root);//bfs层序遍历
    while(!q.empty()){
        Node *tree=q.front();
        q.pop();
        v.push_back(tree->data);
        if(tree->lchild) q.push(tree->lchild);
        if(tree->rchild) q.push(tree->rchild);
    }

    int len=v.size();
    for(int i=0;i<len;i++)//输出
        printf("%s%d",(i==0?"":" "),v[i]);


    return 0;
}
```
