

### 1066 Root of AVL Tree (25分)

* 1，```L，R，LL，LR，RR，RL```

```
struct Node{
    int v,height;
    Node *lf,*rt;
};

Node* newNode(int v){
    Node* node=new Node;
    node->v=v;
    node->height=1;
    node->lf=node->rt=NULL;
    return node;
}

int getHeight(Node* root){
    if(root==NULL)
        return 0;
    return root->height;
}

///计算结点root的平衡因子
int getBalanceFactor(Node* root){
	return getHeight(root->lf)-getHeight(root->rt);
}

///更新结点root的height
void updateHeight(Node* root){
	root->height=max(getHeight(root->lf),getHeight(root->rt))+1;
	///别忘了"+1"操作
}

///左旋(Left Rotation)，向左转
void L(Node* &root){
	Node* temp=root->rt;
	root->rt=temp->lf;
	temp->lf=root;
	updateHeight(root);
	updateHeight(temp);
	root=temp;
}

///右旋(Right Rotation)
void R(Node* &root){
    Node* temp=root->lf;
    root->lf=temp->rt;
    temp->rt=root;
    updateHeight(root);
    updateHeight(temp);
    root=temp;
}

void insert_(Node* &root,int v){
    if(root==NULL){
        root=newNode(v);
        return ;
    }
    if(v<root->v){
        insert_(root->lf,v);
        updateHeight(root);
        if(getBalanceFactor(root)==2){
            if(getBalanceFactor(root->lf)==1){
                R(root);
            }
            else if(getBalanceFactor(root->lf)==-1){
                L(root->lf);
                R(root);
            }
        }
    }
    else{
        insert_(root->rt,v);
        updateHeight(root);
        if(getBalanceFactor(root)==-2){
            if(getBalanceFactor(root->rt)==-1){
                L(root);
            }
            else if(getBalanceFactor(root->rt)==1){
                R(root->rt);
                L(root);
            }
        }
    }
}

```

### 1123 Is It a Complete AVL Tree (30分)

* 判断是不不是完全⼆二叉树，就看在出现了了⼀一个孩⼦子为空的结点之后是否还会出现孩⼦子结点不不为空的结点，如果出现了了就不不是完全⼆二叉树。

```

int isComplete=1,after=0;
vector<int> levelorder(Node* tree){
    vector<int> v;
    queue<Node *> q;
    q.push(tree);

    while(!q.empty()){
        Node* tmp=q.front();
        q.pop();

        v.push_back(tmp->v);
        if(tmp->lf!=NULL){
            if(after)   isComplete=0;
            q.push(tmp->lf);
        }
        else{
            after=1;
        }

        if(tmp->rt!=NULL){
            if(after)   isComplete=0;
            q.push(tmp->rt);
        }
        else{
            after=1;
        }
    }

    return v;
}

```
