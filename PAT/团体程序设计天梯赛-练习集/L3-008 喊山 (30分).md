```
#include <bits/stdc++.h>

using namespace std;

const int maxn=10010;

struct node{
    int height;
    int index;
}nd;

vector<int> e[maxn];
bool vis[maxn];
int n,m,k;

int bfs(int st){
    int curh=0;///当前树高
    int res=st;///结果,结点
    nd.height=0;nd.index=st;
    queue<node> q;
    q.push(nd);

    vis[st]=true;
    while(!q.empty()){
        nd=q.front();
        q.pop();

        int tmph=nd.height;
        int tmp=nd.index;

        if(tmph==curh){///取编号最小的结点
            res=min(res,tmp);
        }
        if(tmph>curh){
            curh=tmph;///更新最大树高
            res=tmp;
        }
        for(int i=0;i<(int)e[tmp].size();i++){
            int u=e[tmp][i];
            if(!vis[u]){
                vis[u]=true;
                nd.height=tmph+1;
                nd.index=u;
                q.push(nd);
            }
        }
    }
    if(curh==0)
        res=0;
    return res;
}

int main(){
    scanf("%d%d%d",&n,&m,&k);
    int u,v;

    for(int i=0;i<m;i++){
        scanf("%d%d",&u,&v);
        e[u].push_back(v);
        e[v].push_back(u);
        ///e[u][v]=e[v][u]=1;
    }

    int tmp;
    for(int q=0;q<k;q++){
        scanf("%d",&tmp);
        memset(vis,false,sizeof(vis));
        int res=bfs(tmp);

        printf("%d\n",res);
    }

    return 0;
}
```

* ```vis```数组，既标识结点是否访问，又记录访问结点深度
```
#include <bits/stdc++.h>

using  namespace std;

const int maxn=10010;
vector<int> s[maxn];
int vis[maxn];
int st;
queue<int> q;

void bfs(){
	int u;
	while(!q.empty()) q.pop();
	q.push(st);
	vis[st]=1;
	while(!q.empty()){
		u=q.front();
		q.pop();
		for(int i=0;i<(int)s[u].size();i++){
			int v=s[u][i];
			if(!vis[v]){
				vis[v]=vis[u]+1;
				q.push(v);
			}
		}
	}
}
int main(){
	int n,m,k,u,v;
	//freopen("t.txt","r",stdin);
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;i++) s[i].clear();
	while(m--){
		scanf("%d%d",&u,&v);
		s[u].push_back(v);
		s[v].push_back(u);
	}
	while(k--){
		scanf("%d",&st);
		memset(vis,0,sizeof(vis));
		bfs();
		int ans=0,r;
		for(int i=1;i<=n;i++){
			if(vis[i]>ans){
				r=i;ans=vis[i];
			}
		}
		if(r!=st)
			printf("%d\n",r);
		else
			printf("0\n");
	}
	return 0;
}
```
